@0x05   0x01    // right 0001

@0x08   0x6E    // X Boundary =110
@0x09   0x28    // Y Boundary 40
@0x0A   0x50    // Y Boundary =80
@0x0B   0x00    //0

@0x0D	0x9F	//Constant 159
@0x0E	0x00	//Vertical counter for VGA
@0x0F	0x77	//Constant 119
@0x10	0x00	//Constant 0
@0x11	0x01	//Constant 1

@0x13	0x3C	//Previous mouse Y data
